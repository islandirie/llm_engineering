# using miniconda
mkdir -p ~/miniconda3
curl https://repo.anaconda.com/miniconda/Miniconda3-latest-MacOSX-arm64.sh -o ~/miniconda3/miniconda.sh
bash ~/miniconda3/miniconda.sh -b -u -p ~/miniconda3
rm ~/miniconda3/miniconda.sh

# after installation, close / reopen terminal. should see (base)
source ~/miniconda3/bin/activate

# initialize conda on all available shells
conda init --all

# go to llm-engineering folder (proj root directory)
conda env create -f environment.yml

#run env, should see (llms). Means you are running correct version of anaconda

# conda deactivate

# If starting from scratch, run "conda activate llms", then, run jupyter lab
jupyter lab
